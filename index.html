Note: You have to complete countCharacters function. No need to take any input. 

        
            let s='bAaDAAbAD'
            function countCharacters(){
                let count0 =0
                let count1=0
                let b=[count0,count1]
                for (const char of s){
                    if (char==="A"){
                        b[0] = count0+=1
